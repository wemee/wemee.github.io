兩組ga4一起設定

<script>
    window.dataLayer = window.dataLayer || [];
    function gtag(){dataLayer.push(arguments);}
    gtag('js', new Date()); // 設定一次時間給兩組共用

    // 同時設定兩組GA4
    gtag('config', 'G-6JCG6ZWMG2');
    gtag('config', 'G-4JQ44BZSW0');
</script>
